\newcommand{\assgrade}[1]{%
\textbf{
    \IfStrEqCase{#1}{%
        {A}{\TR{Extraordinarily challenging}{Mimořádně náročné}}%
        {B}{\TR{Challenging}{Náročné}}%
        {C}{\TR{Ordinarily challenging}{Průměrně náročné}}%
        {D}{\TR{Easy}{Lehčí}}%
        {E}{\TR{Unsatisfactorily challenging}{Nedostatečně náročné}}%
    }[{\TR{Invalid Grade}{Neplatná známka}}]%
}
}

\newcommand{\fullfilgrade}[1]{%
\textbf{
    \IfStrEqCase{#1}{%
        {A}{\TR{Fulfilled}{Splněno}}%
        {B}{\TR{Fulfilled with minor objections}{Splněno s menšími výhradami}}%
        {C}{\TR{Fulfilled with major objections}{Splněno s většími výhradami}}%
        {D}{\TR{Unfulfilled}{Nesplněno}}%
    }[{\TR{Invalid Grade}{Neplatná známka}}]%
}
}



\newcommand{\methodologygrade}[1]{
  \textbf{
    \IfStrEqCase{#1}{%
        {A}{\TR{Outstanding}{Vynikající}}%
        {B}{\TR{Correct}{Správný}}%
        {C}{\TR{Partially applicable}{Částečně vhodný}}%
        {D}{\TR{Incorrect}{Nesprávný}}% 
    }[{\TR{Invalid Grade}{Neplatná známka}}]%
  }
}



\newcommand{\generalgrade}[1]{%
\textbf{
    \IfStrEqCase{#1}{%
        {A}{\TR{A - Excellent}{A - Výborně}}%
        {B}{\TR{B - Very good}{B - Velmi dobře}}%
        {C}{\TR{C - Good}{C - Dobře}}%
        {D}{\TR{D - Satisfactory}{D - Uspokojivě}}%
        {E}{\TR{E - Sufficient}{E - Dostatčně}}%
        {F}{\TR{F - Failed}{F - Nedostatečně}}%
    }[{\TR{Invalid Grade}{Neplatná známka}}]%
}
}
